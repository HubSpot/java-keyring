27  package com.github.keyring.example;
28  
29  import java.io.File;
30  import java.io.IOException;
31  import java.util.logging.Level;
32  import java.util.logging.Logger;
33  
34  import com.github.javakeyring.BackendNotSupportedException;
35  import com.github.javakeyring.Keyring;
36  import com.github.javakeyring.PasswordRetrievalException;
37  import com.github.javakeyring.PasswordSaveException;
38  import com.github.javakeyring.util.LockException;
39  
40  /**
41   * Usage example of java-keyring library.
42   */
43  public class Program {
44  
45    /**
46     * Sample main method.
47     * 
48     * @param args
49     *          the command line arguments.
50     */
51    public static void main(String[] args) {
52  
53      //
54      // setup a Keyring instance
55      //
56      Keyring keyring;
57  
58      // create an instance of Keyring by invoking Keyring.create method
59      //
60      // Keyring.create method finds appropriate keyring backend, and sets it up
61      // for you.
62      // On Mac OS X environment, OS X Keychain is used, and On Windows
63      // environment,
64      // DPAPI is used for encryption of passwords.
65      // If no supported backend is found, BackendNotSupportedException is thrown.
66      try {
67        keyring = Keyring.create();
68      } catch (BackendNotSupportedException ex) {
69        Logger.getLogger(Program.class.getName()).log(Level.SEVERE, null, ex);
70        return;
71      }
72  
73      // some backend directory handles a file to store password to disks.
74      // in this case, we must set path to key store file by
75      // Keyring.setKeyStorePath
76      // before using Keyring.getPassword and Keyring.getPassword.
77      if (keyring.isKeyStorePathRequired()) {
78        try {
79          File keyStoreFile = File.createTempFile("keystore", ".keystore");
80          keyring.setKeyStorePath(keyStoreFile.getPath());
81        } catch (IOException ex) {
82          Logger.getLogger(Program.class.getName()).log(Level.SEVERE, null, ex);
83        }
84      }
85  
86      //
87      // store password to key store
88      //
89  
90      // Password can be stored to key store by using Keyring.setPassword method.
91      // PasswordSaveException is thrown when some error happened while saving
92      // password.
93      // LockException is thrown when keyring backend failed to lock key store
94      // file.
95      try {
96        keyring.setPassword("My service name", "My account name", "My password");
97      } catch (LockException ex) {
98        Logger.getLogger(Program.class.getName()).log(Level.SEVERE, null, ex);
99        return;
100     } catch (PasswordSaveException ex) {
101       Logger.getLogger(Program.class.getName()).log(Level.SEVERE, null, ex);
102       return;
103     }
104 
105     //
106     // Retrieve password from key store
107     //
108 
109     // Password can be retrieved by using Keyring.getPassword method.
110     // PasswordRetrievalException is thrown when some error happened while
111     // getting password.
112     // LockException is thrown when keyring backend failed to lock key store
113     // file.
114     try {
115       String password = keyring.getPassword("My service name", "My account name");
116       System.out.println(password);
117     } catch (LockException ex) {
118       Logger.getLogger(Program.class.getName()).log(Level.SEVERE, null, ex);
119     } catch (PasswordRetrievalException ex) {
120       Logger.getLogger(Program.class.getName()).log(Level.SEVERE, null, ex);
121     }
122   }
123 } // class Program
